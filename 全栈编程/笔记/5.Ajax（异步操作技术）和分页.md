	# 对提交进行事件监听
	$('#id').click(function(){
		# 获取前端的内容
		let a = $('id').val()
		
		#用ajax把数据传送给后端views
		#里面数据传入的是对象的形式
		$.ajax({
			url:'',
			type:'post',
			data:{数据}，
			success:function(){
				# 数据成功传给后端后，后端把数据传给这里
			},
			error:function(){
				alert('数据传输失败!')
			}
		})
	})
<script/>
